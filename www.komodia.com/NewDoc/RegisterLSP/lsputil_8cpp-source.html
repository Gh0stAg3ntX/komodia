00370 //
00371 // Description:
00372 //      This routne counts how many orphaned layered protocol entries exist. An 
00373 //      orphaned protocol entry is a protocol entry whose chain length greater
00374 //      than one and whose dummy hidden entry (i.e. index 0 of its protocol chain
00375 //      array) is missing. When building the LSP map, it normally finds entries

00380 CountOrphanedChainEntries(
00381     WSAPROTOCOL_INFOW  *pCatalog,
00382     int                 iCatalogCount
00383     )
00384 {
00385     int     orphanCount = 0,
00386             i, j;
00387 
00388     for(i=0; i < iCatalogCount ;i++)
00389     {
00390         if ( pCatalog[ i ].ProtocolChain.ChainLen > 1 )
00391         {
00392             for(j=0; j < iCatalogCount ;j++)
00393             {
00394                 if ( i == j )
00395                     continue;
00396                 if ( pCatalog[ j ].dwCatalogEntryId == pCatalog[ i ].ProtocolChain.ChainEntries[ 0 ] )
00397                 {
00398                     break;
00399                 }
00400             }
00401             if ( j >= iCatalogCount )
00402                 orphanCount++;
00403         }
00404     }
